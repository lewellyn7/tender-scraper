## 一、现状问题分析

### 1.1 核心瓶颈

| 问题 | 现状 | 影响 |
|------|------|------|
| **同步阻塞** | `requests.get()` 阻塞等待响应 | 并发能力受限，CPU空闲 |
| **存储层简陋** | SQLite + Excel，无索引 | 查询慢，无法支撑大数据量 |
| **重试策略粗犷** | 固定次数 + 简单 sleep | 封禁时仍盲目重试 |
| **无缓存层** | 每次采集重新获取 | 浪费带宽，易触发反爬 |
| **状态管理缺失** | 无连接池复用 | TCP握手开销大 |

### 1.2 同步 vs 异步性能对比

```
同步采集 100 个 URL（假设平均响应 500ms）：
  总耗时 = 100 × 500ms = 50,000ms = 50秒

异步采集 100 个 URL（并发 20）：
  总耗时 = (100/20) × 500ms = 2,500ms = 2.5秒
  提升 = 20倍
```

#### 2.2.1 aiohttp Session 管理

```python
# app/crawler/async_session.py
import asyncio
import aiohttp
from typing import Optional
from contextlib import asynccontextmanager
from dataclasses import dataclass
import logging

logger = logging.getLogger(__name__)

@dataclass
class RateLimitConfig:
    requests_per_second: float = 5.0
    burst_size: int = 10

class AsyncCrawlerSession:
    """
    异步采集 Session 管理器
    - 连接池复用
    - 令牌桶限速
    - 自动重试
    """

    def __init__(
        self,
        rate_limit: RateLimitConfig = None,
        timeout: int = 30,
        max_retries: int = 3
    ):
        self.rate_limit = rate_limit or RateLimitConfig()
        self.timeout = aiohttp.ClientTimeout(total=timeout)
        self.max_retries = max_retries
        self._session: Optional[aiohttp.ClientSession] = None
        self._semaphore = asyncio.Semaphore(20)  # 最大并发20
        self._token_bucket = TokenBucket(
            rate=self.rate_limit.requests_per_second,
            capacity=self.rate_limit.burst_size
        )

    async def __aenter__(self):
        connector = aiohttp.TCPConnector(
            limit=100,           # 连接池上限
            limit_per_host=10,   # 单host并发上限
            ttl_dns_cache=300,   # DNS缓存5分钟
            enable_cleanup_closed=True,
        )
        self._session = aiohttp.ClientSession(
            connector=connector,
            timeout=self.timeout,
            headers={
                "User-Agent": "Mozilla/5.0 (compatible; TenderBot/3.0)",
                "Accept-Encoding": "gzip, deflate",
            }
        )
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        if self._session:
            await self._session.close()

    async def fetch(
        self,
        url: str,
        method: str = "GET",
        **kwargs
    ) -> Optional[str]:
        """带限速和重试的请求"""
        await self._token_bucket.acquire()

        async with self._semaphore:
            for attempt in range(self.max_retries):
                try:
                    async with self._session.request(
                        method, url, **kwargs
                    ) as response:
                        if response.status == 200:
                            return await response.text()
                        elif response.status == 429:
                            # 限流，等待更长时间
                            wait_time = int(response.headers.get(
                                "Retry-After", 60
                            ))
                            logger.warning(f"Rate limited, waiting {wait_time}s")
                            await asyncio.sleep(wait_time)
                        elif response.status >= 500:
                            # 服务器错误，重试
                            await asyncio.sleep(2 ** attempt)
                        else:
                            return None
                except asyncio.TimeoutError:
                    logger.warning(f"Timeout for {url}, attempt {attempt+1}")
                    await asyncio.sleep(2 ** attempt)
                except Exception as e:
                    logger.error(f"Request failed: {e}")
                    if attempt == self.max_retries - 1:
                        return None
                    await asyncio.sleep(2 ** attempt)

        return None


class TokenBucket:
    """令牌桶算法实现"""

    def __init__(self, rate: float, capacity: int):
        self.rate = rate
        self.capacity = capacity
        self.tokens = capacity
        self.last_update = asyncio.get_event_loop().time()
        self._lock = asyncio.Lock()

    async def acquire(self):
        async with self._lock:
            while self.tokens < 1:
                await self._refill()
                await asyncio.sleep(0.1)
            self.tokens -= 1

    async def _refill(self):
        now = asyncio.get_event_loop().time()
        elapsed = now - self.last_update
        self.tokens = min(
            self.capacity,
            self.tokens + elapsed * self.rate
        )
        self.last_update = now
```

#### 2.2.2 批量采集任务

```python
# app/crawler/batch_collector.py
import asyncio
from typing import List, Callable, Any
from dataclasses import dataclass
import logging

logger = logging.getLogger(__name__)

@dataclass
class CrawlTask:
    url: str
    priority: int = 5
    source: str = ""
    metadata: dict = None

    def __lt__(self, other):
        return self.priority > other.priority  # 优先级高的先执行

class BatchCollector:
    """
    批量采集器
    - 优先级队列
    - 批量并发控制
    - 结果聚合
    """

    def __init__(
        self,
        session: AsyncCrawlerSession,
        concurrency: int = 20,
        priority_enabled: bool = True
    ):
        self.session = session
        self.concurrency = concurrency
        self.priority_enabled = priority_enabled
        self.queue: asyncio.PriorityQueue = None

    async def collect(
        self,
        tasks: List[CrawlTask],
        progress_callback: Callable[[int, int], None] = None
    ) -> List[dict]:
        """批量采集，支持优先级"""

        if self.priority_enabled:
            # 使用优先级队列
            self.queue = asyncio.PriorityQueue()
            for task in tasks:
                await self.queue.put(task)
        else:
            # 使用普通队列
            self.queue = asyncio.Queue()
            for task in tasks:
                await self.queue.put(task)

        results = []
        completed = 0
        total = len(tasks)

        # 创建并发控制信号量
        semaphore = asyncio.Semaphore(self.concurrency)

        async def worker():
            nonlocal completed
            while True:
                try:
                    task = self.queue.get_nowait()
                except asyncio.QueueEmpty:
                    break

                async with semaphore:
                    result = await self._crawl_single(task)
                    if result:
                        results.append(result)

                    completed += 1
                    if progress_callback:
                        progress_callback(completed, total)

                    self.queue.task_done()

        # 启动 workers
        workers = [
            asyncio.create_task(worker())
            for _ in range(min(self.concurrency, total))
        ]

        await asyncio.gather(*workers, return_exceptions=True)

        return results

    async def _crawl_single(self, task: CrawlTask) -> dict:
        """采集单个任务"""
        content = await self.session.fetch(task.url)

        if not content:
            return None

        return {
            "url": task.url,
            "source": task.source,
            "content": content,
            "metadata": task.metadata or {},
            "crawled_at": datetime.now().isoformat(),
        }
```

### 3.2 Redis 缓存策略

```python
# app/cache/redis_cache.py
import redis.asyncio as redis
import json
from typing import Optional, Any
from datetime import timedelta
import hashlib

class RedisCache:
    """Redis 缓存层"""

    CRAWL_PREFIX = "crawl:"
    RATE_LIMIT_PREFIX = "ratelimit:"
    LOCK_PREFIX = "lock:"

    def __init__(self, redis_url: str = "redis://localhost:6379/0"):
        self.redis = redis.from_url(redis_url, decode_responses=True)

    async def get_cached(self, url: str) -> Optional[str]:
        """获取缓存的页面内容"""
        key = f"{self.CRAWL_PREFIX}{self._hash_key(url)}"
        return await self.redis.get(key)

    async def set_cached(
        self,
        url: str,
        content: str,
        ttl: int = 3600  # 1小时
    ):
        """缓存页面内容"""
        key = f"{self.CRAWL_PREFIX}{self._hash_key(url)}"
        await self.redis.setex(key, timedelta(seconds=ttl), content)

    async def is_rate_limited(self, domain: str) -> bool:
        """检查是否被限流"""
        key = f"{self.RATE_LIMIT_PREFIX}{domain}"
        return await self.redis.exists(key) > 0

    async def set_rate_limited(
        self,
        domain: str,
        wait_seconds: int = 3600
    ):
        """标记限流"""
        key = f"{self.RATE_LIMIT_PREFIX}{domain}"
        await self.redis.setex(key, timedelta(seconds=wait_seconds), "1")

    async def acquire_lock(
        self,
        lock_name: str,
        timeout: int = 30
    ) -> bool:
        """获取分布式锁"""
        key = f"{self.LOCK_PREFIX}{lock_name}"
        return await self.redis.set(
            key, "1", nx=True, ex=timeout
        )

    async def release_lock(self, lock_name: str):
        """释放分布式锁"""
        key = f"{self.LOCK_PREFIX}{lock_name}"
        await self.redis.delete(key)

    @staticmethod
    def _hash_key(url: str) -> str:
        return hashlib.md5(url.encode()).hexdigest()[:16]
```

### 4.1 智能重试策略

```python
# app/crawler/retry_strategies.py
from dataclasses import dataclass
from enum import Enum
import asyncio

class AnomalyType(Enum):
    NETWORK_TIMEOUT = "network_timeout"
    RATE_LIMIT = "rate_limit"
    BAN = "ban"
    SERVER_ERROR = "server_error"
    PARSE_ERROR = "parse_error"
    UNKNOWN = "unknown"

@dataclass
class RetryStrategy:
    anomaly_type: AnomalyType
    base_delay: float       # 基础等待时间(秒)
    max_delay: float        # 最大等待时间
    multiplier: float       # 退避倍数
    max_retries: int        # 最大重试次数

RETRY_STRATEGIES = {
    AnomalyType.NETWORK_TIMEOUT: RetryStrategy(
        anomaly_type=AnomalyType.NETWORK_TIMEOUT,
        base_delay=2,
        max_delay=30,
        multiplier=2,
        max_retries=3
    ),
    AnomalyType.RATE_LIMIT: RetryStrategy(
        anomaly_type=AnomalyType.RATE_LIMIT,
        base_delay=60,       # 较长等待
        max_delay=900,       # 最多等15分钟
        multiplier=1.5,
        max_retries=5
    ),
    AnomalyType.BAN: RetryStrategy(
        anomaly_type=AnomalyType.BAN,
        base_delay=300,      # 直接等5分钟
        max_delay=3600,
        multiplier=1.0,      # 线性等待
        max_retries=2
    ),
    AnomalyType.SERVER_ERROR: RetryStrategy(
        anomaly_type=AnomalyType.SERVER_ERROR,
        base_delay=5,
        max_delay=60,
        multiplier=2,
        max_retries=3
    ),
    AnomalyType.PARSE_ERROR: RetryStrategy(
        anomaly_type=AnomalyType.PARSE_ERROR,
        base_delay=0,        # 不等待，立即重试
        max_delay=0,
        multiplier=0,
        max_retries=1
    ),
}

class SmartRetry:
    """
    智能重试器
    - 根据异常类型选择策略
    - 指数退避 + 抖动
    - 记录失败历史
    """

    def __init__(self):
        self.failure_history: dict[str, list] = {}  # url -> [attempts]

    def classify_error(
        self,
        error: Exception,
        response_status: int = None,
        response_text: str = None
    ) -> AnomalyType:
        """分类异常类型"""
        error_msg = str(error).lower()

        # Rate limit
        if response_status == 429 or "rate limit" in error_msg:
            return AnomalyType.RATE_LIMIT

        # Ban
        if response_status == 403 or "forbidden" in error_msg:
            return AnomalyType.BAN

        # Server error
        if response_status and 500 <= response_status < 600:
            return AnomalyType.SERVER_ERROR

        # Timeout
        if "timeout" in error_msg or "timed out" in error_msg:
            return AnomalyType.NETWORK_TIMEOUT

        # Parse error
        if "parse" in error_msg or "html" in error_msg:
            return AnomalyType.PARSE_ERROR

        return AnomalyType.UNKNOWN

    async def execute_with_retry(
        self,
        task: Callable,
        url: str,
        anomaly_type: AnomalyType = None,
        **kwargs
    ):
        """执行任务并重试"""
        strategy = RETRY_STRATEGIES.get(
            anomaly_type or AnomalyType.UNKNOWN
        )

        last_error = None
        for attempt in range(strategy.max_retries):
            try:
                result = await task(**kwargs)
                # 成功，清除历史
                self._clear_history(url)
                return result

            except Exception as e:
                last_error = e
                if attempt < strategy.max_retries - 1:
                    delay = self._calculate_delay(
                        strategy, attempt
                    )
                    await asyncio.sleep(delay)

        # 最终失败，记录
        self._record_failure(url, anomaly_type)
        raise last_error

    def _calculate_delay(
        self,
        strategy: RetryStrategy,
        attempt: int
    ) -> float:
        """计算延迟：指数退避 + 抖动"""
        if strategy.base_delay == 0:
            return 0

        # 指数退避
        delay = strategy.base_delay * (strategy.multiplier ** attempt)

        # 抖动 ±25%
        import random
        jitter = delay * 0.25 * (2 * random.random() - 1)

        return min(delay + jitter, strategy.max_delay)

    def _record_failure(self, url: str, anomaly_type: AnomalyType):
        """记录失败历史"""
        if url not in self.failure_history:
            self.failure_history[url] = []
        self.failure_history[url].append({
            "type": anomaly_type.value,
            "timestamp": datetime.now().isoformat()
        })

    def _clear_history(self, url: str):
        """清除历史"""
        self.failure_history.pop(url, None)
```

### 5.1 多级限流

```python
# app/crawler/rate_limiter.py
import asyncio
import time
from dataclasses import dataclass
from typing import Dict

@dataclass
class RateLimitConfig:
    """限流配置"""
    requests_per_second: float   # 全局每秒请求数
    requests_per_minute: float   # 全局每分钟请求数
    per_host_rps: float          # 单host每秒请求数
    per_host_rpm: float          # 单host每分钟请求数

class MultiLevelRateLimiter:
    """
    多级限流器
    - 全局限流
    - 单host限流
    - 突发流量控制
    """

    def __init__(self, config: RateLimitConfig):
        self.config = config

        # 全局计数器
        self._global_rps = TokenBucket(config.requests_per_second, 100)
        self._global_rpm = TokenBucket(config.requests_per_minute, 1000)

        # 单host计数器
        self._host_counters: Dict[str, dict] = {}

        # 信号量控制并发
        self._semaphore = asyncio.Semaphore(20)

    async def acquire(self, host: str):
        """获取请求许可"""
        # 全局限流
        await self._global_rps.acquire()
        await self._global_rpm.acquire()

        # 单host限流
        await self._get_host_limiter(host).acquire()

        # 并发控制
        await self._semaphore.acquire()

    def release(self, host: str):
        """释放信号量"""
        self._semaphore.release()

    def _get_host_limiter(self, host: str) -> TokenBucket:
        """获取host专属限流器"""
        if host not in self._host_counters:
            self._host_counters[host] = {
                "rps": TokenBucket(
                    self.config.per_host_rps,
                    self.config.per_host_rps * 2
                ),
                "rpm": TokenBucket(
                    self.config.per_host_rpm,
                    self.config.per_host_rpm * 2
                ),
            }
        return self._host_counters[host]["rps"]


class TokenBucket:
    """令牌桶"""

    def __init__(self, rate: float, capacity: int):
        self.rate = rate
        self.capacity = capacity
        self.tokens = capacity
        self.last_update = time.monotonic()
        self._lock = asyncio.Lock()

    async def acquire(self):
        async with self._lock:
            self._refill()
            while self.tokens < 1:
                await asyncio.sleep(0.01)
                self._refill()
            self.tokens -= 1

    def _refill(self):
        now = time.monotonic()
        elapsed = now - self.last_update
        self.tokens = min(
            self.capacity,
            self.tokens + elapsed * self.rate
        )
        self.last_update = now
```
